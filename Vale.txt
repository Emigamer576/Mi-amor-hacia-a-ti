        const audioCtx = new (window.AudioContext || window.webkitAudioContext)();

        function playTone(freq1, freq2, attack, dur, type = 'sine') {
            if(audioCtx.state === 'suspended') audioCtx.resume();
            const osc = audioCtx.createOscillator(), gain = audioCtx.createGain();
            osc.type = type; 
            osc.frequency.setValueAtTime(freq1, audioCtx.currentTime); 
            osc.frequency.exponentialRampToValueAtTime(freq2, audioCtx.currentTime + dur);
            gain.gain.setValueAtTime(0, audioCtx.currentTime); 
            gain.gain.linearRampToValueAtTime(0.15, audioCtx.currentTime + attack); 
            gain.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + dur);
            osc.connect(gain); gain.connect(audioCtx.destination); 
            osc.start(); osc.stop(audioCtx.currentTime + dur);
        }

        // Función para subir el volumen de Epic gradualmente
            function fadeInEpicAudio() {
            const audioEpic = document.getElementById('audio-epic');
            const audioInicio = document.getElementById('audio-inicio');
            if(!audioEpic) return;
            
            epicAudioFadedIn = true;

            // TRUCO: Igualamos el tiempo exacto de Epic con el de Inicio 
            // para garantizar que la pista no esté desfasada.
            if(audioInicio && audioInicio.currentTime > 0) {
                audioEpic.currentTime = audioInicio.currentTime;
            }
            
            let vol = 0;
            // Sube el volumen 0.02 cada 200ms -> tarda 10 segundos en llegar a 1.0 (volumen máximo)
            const fadeInterval = setInterval(() => {
                if (vol < 1.0) {
                    vol += 0.02;
                    audioEpic.volume = Math.min(vol, 1.0);
                } else {
                    clearInterval(fadeInterval);
                }
            }, 50);
        }

        async function startFlashbackSequence(index) {
            const overlay = document.getElementById('flashback-overlay');
            const textContainer = document.getElementById('flashback-text');
            const talkAudio = document.getElementById('audio-talk');
            
            sfx.transport();
            overlay.style.display = 'flex';
            setTimeout(() => overlay.style.opacity = '1', 50);
            
            await sleep(3000); 
            
            textContainer.innerHTML = unlockedParagraph;
            
            const newSnippet = storySnippets[index] || " (Texto oculto revelado).";
            
            for (let i = 0; i < newSnippet.length; i++) {
                textContainer.innerHTML += newSnippet.charAt(i);
                
                if (newSnippet.charAt(i) !== ' ') {
                    if(talkAudio) {
                        talkAudio.currentTime = 0;
                        talkAudio.play().catch(() => {});
                    }
                }
                await sleep(50); 
            }

            unlockedParagraph += newSnippet;

            await sleep(3000); 

            overlay.style.opacity = '0';
            await sleep(1000);
            overlay.style.display = 'none';
            textContainer.innerHTML = ""; 
            
            currentLevel++;
            renderTimeline();
            
            if(currentLevel < storyEvents.length) {
                const next = document.getElementById(`level-${currentLevel}`);
                next.scrollIntoView({ behavior: 'smooth', block: 'center' });
                setTimeout(() => { next.classList.add('show-anim'); sfx.pop(); }, 500);
            }
        }

        function initSimon() {
            const colors = ['#e91e63', '#2196f3', '#4caf50', '#ffc107'];
            gameArea.innerHTML = `<div id="simon-board" style="display:grid; grid-template-columns: 1fr 1fr;"></div>`;
            const board = document.getElementById('simon-board');
            let sequence = [], playerSeq = [];
            colors.forEach((color, i) => {
                const btn = document.createElement('button'); btn.className = 'simon-btn'; btn.style.backgroundColor = color;
                btn.onclick = () => handleSimonClick(i); board.appendChild(btn);
            });
            function nextRound() { playerSeq = []; sequence.push(Math.floor(Math.random() * 4)); playSequence(); }
            function playSequence() {
                let i = 0; board.style.pointerEvents = 'none';
                const interval = setInterval(() => {
                    board.children[sequence[i]].classList.add('active'); sfx.pop();
                    setTimeout(() => board.children[sequence[i-1]].classList.remove('active'), 400);
                    i++; if (i >= sequence.length) { clearInterval(interval); board.style.pointerEvents = 'auto'; }
                }, 800);
            }
            function handleSimonClick(index) {
                board.children[index].classList.add('active'); sfx.click();
                setTimeout(() => board.children[index].classList.remove('active'), 200);
                playerSeq.push(index);
                if (playerSeq[playerSeq.length - 1] !== sequence[playerSeq.length - 1]) { sfx.error(); feedback.textContent = '¡Error! Reiniciando...'; sequence = []; setTimeout(nextRound, 1000); return; }
                if (playerSeq.length === sequence.length) { if (sequence.length === 4) winLevel(); else setTimeout(nextRound, 1000); }
            }
            setTimeout(nextRound, 500);
        }

        function initCeleste() {
            gameArea.innerHTML = `<canvas id="celesteCanvas" width="400" height="300"></canvas>`; const ctx = document.getElementById('celesteCanvas').getContext('2d');
            let p = { x: 20, y: 250, vx: 0, vy: 0, w: 15, h: 15, speed: 3, jump: -6, dash: 8 }, st = { gr: false, dash: true, isD: false, dT: 0, fr: true }, keys = {};
            window.onkeydown = e => keys[e.code] = true; window.onkeyup = e => keys[e.code] = false;
            const plats = [ [0, 280, 400, 20], [100, 230, 50, 10], [200, 180, 50, 10], [300, 100, 20, 100], [150, 80, 50, 10], [20, 60, 40, 10] ], crys = [{x: 270, y: 150, act: true}], goal = {x: 25, y: 35, w: 20, h: 25};
            function update() {
                if(!isGameRunning) return;
                if (!st.isD) {
                    if (keys['ArrowLeft']) { p.vx = -p.speed; st.fr = false; } else if (keys['ArrowRight']) { p.vx = p.speed; st.fr = true; } else p.vx *= 0.8;
                    p.vy += 0.3; if ((keys['KeyZ'] || keys['ArrowUp']) && st.gr) { p.vy = p.jump; st.gr = false; keys['KeyZ'] = keys['ArrowUp'] = false; sfx.jump(); }
                    if (keys['KeyX'] && st.dash) {
                        st.isD = true; st.dash = false; st.dT = 15; let dx=0, dy=0;
                        if(keys['ArrowRight']) dx=1; else if(keys['ArrowLeft']) dx=-1; if(keys['ArrowUp']) dy=-1; else if(keys['ArrowDown']) dy=1;
                        if(!dx && !dy) dx = st.fr?1:-1; if(dx && dy) { dx*=0.707; dy*=0.707; } p.vx=dx*p.dash; p.vy=dy*p.dash; keys['KeyX']=false;
                    }
                } else { st.dT--; if (st.dT <= 0) { st.isD = false; p.vx*=0.5; p.vy*=0.5; } }
                p.x += p.vx; p.y += p.vy; st.gr = false;
                if(p.x<0) p.x=0; if(p.x+p.w>400) p.x=400-p.w; if(p.y>300) { sfx.error(); p.x=20; p.y=250; p.vy=0; }
                for(let pl of plats) {
                    if (p.x < pl[0]+pl[2] && p.x+p.w > pl[0] && p.y < pl[1]+pl[3] && p.y+p.h > pl[1]) {
                        let px = p.x-p.vx, py = p.y-p.vy;
                        if (py+p.h <= pl[1]) { p.y=pl[1]-p.h; p.vy=0; st.gr=st.dash=true; } else if (py >= pl[1]+pl[3]) { p.y=pl[1]+pl[3]; p.vy=0; } else if (px+p.w <= pl[0]) { p.x=pl[0]-p.w; p.vx=0; } else if (px >= pl[0]+pl[2]) { p.x=pl[0]+pl[2]; p.vx=0; }
                    }
                }
                crys.forEach(c => { if(c.act && p.x<c.x+10 && p.x+p.w>c.x && p.y<c.y+10 && p.y+p.h>c.y) { sfx.catch(); c.act=false; st.dash=true; setTimeout(()=>c.act=true, 3000); } });
                ctx.fillStyle = '#111'; ctx.fillRect(0,0,400,300);
                ctx.fillStyle = '#455a64'; plats.forEach(pl => { ctx.fillRect(pl[0], pl[1], pl[2], pl[3]); ctx.fillStyle='#cfd8dc'; ctx.fillRect(pl[0], pl[1], pl[2], 3); ctx.fillStyle='#455a64'; });
                crys.forEach(c => { if(c.act) { ctx.fillStyle='#00e5ff'; ctx.fillRect(c.x, c.y, 10, 10); } });
                ctx.fillStyle = '#f06292'; ctx.fillRect(goal.x, goal.y, goal.w, goal.h);
                ctx.fillStyle = st.dash ? '#f44336' : '#2196f3'; ctx.fillRect(p.x, p.y, p.w, p.h);
                if(st.isD) { ctx.fillStyle='rgba(244,67,54,0.5)'; ctx.fillRect(p.x-p.vx, p.y-p.vy, p.w, p.h); }
                if(p.x < goal.x+goal.w && p.x+p.w > goal.x && p.y < goal.y+goal.h && p.y+p.h > goal.y) winLevel(); else animationFrameId = requestAnimationFrame(update);
            }
            update();
        }

        const canvasParticles = document.getElementById('particles-canvas');
        const ctxParticles = canvasParticles.getContext('2d');
        let particlesArray = [];
        
        function resizeCanvas() {
            canvasParticles.width = window.innerWidth;
            canvasParticles.height = window.innerHeight;
        }
        window.addEventListener('resize', resizeCanvas);
        resizeCanvas();

        class Particle {
            constructor() {
                this.reset();
            }
            reset() {
                this.x = Math.random() * canvasParticles.width;
                this.y = Math.random() * canvasParticles.height - canvasParticles.height;
                this.size = Math.random() * 6 + 2;
                this.speedY = Math.random() * 1 + 0.5;
                this.speedX = Math.random() * 2 - 1;
                this.angle = Math.random() * 360;
                this.spin = Math.random() * 5 - 2.5;
                // Tipo 0: Pétalo Sakura (hoja), Tipo 1: Orbe brillante (algo bonito extra)
                this.type = Math.random() > 0.4 ? 0 : 1; 
                this.opacity = Math.random() * 0.6 + 0.4;
            }
            update() {
                // La velocidad aumenta ligeramente según el nivel
                let levelMultiplier = 1 + (currentLevel * 0.2);
                
                this.y += this.speedY * levelMultiplier;
                this.x += (this.speedX + Math.sin(this.angle * Math.PI / 180)) * (levelMultiplier * 0.6);
                this.angle += this.spin;

                if (this.y > canvasParticles.height) {
                    this.reset();
                    this.y = -10; 
                }
            }
            draw() {
                ctxParticles.save();
                ctxParticles.translate(this.x, this.y);
                ctxParticles.rotate(this.angle * Math.PI / 180);
                
                if (this.type === 0) {
                    // Pétalo de Sakura
                    ctxParticles.fillStyle = `rgba(255, 183, 206, ${this.opacity})`;
                    ctxParticles.beginPath();
                    ctxParticles.ellipse(0, 0, this.size, this.size / 2, 0, 0, Math.PI * 2);
                    ctxParticles.fill();
                } else {
                    // Orbe de luz mágico
                    ctxParticles.beginPath();
                    let gradient = ctxParticles.createRadialGradient(0, 0, 0, 0, 0, this.size);
                    gradient.addColorStop(0, `rgba(255, 255, 255, ${this.opacity})`);
                    gradient.addColorStop(1, `rgba(255, 255, 255, 0)`);
                    ctxParticles.fillStyle = gradient;
                    ctxParticles.arc(0, 0, this.size, 0, Math.PI * 2);
                    ctxParticles.fill();
                }
                ctxParticles.restore();
            }
        }

        function initParticles() {
            particlesArray = [];
            for (let i = 0; i < 70; i++) {
                particlesArray.push(new Particle());
            }
        }

        function animateParticles() {
            ctxParticles.clearRect(0, 0, canvasParticles.width, canvasParticles.height);
            // Mostrar partículas solo si ya empezamos (si currentLevel > 0 o la intro terminó)
            if (document.getElementById('intro-screen').style.display === 'none') {
               for (let i = 0; i < particlesArray.length; i++) {
                   // Aumenta la cantidad de partículas visibles según el nivel (del 20% al 100%)
                   let targetCount = Math.floor(70 * (0.2 + (currentLevel * 0.1)));
                   if(i < targetCount) {
                       particlesArray[i].update();
                       particlesArray[i].draw();
                   }
               }
            }
            requestAnimationFrame(animateParticles);
        }

        initParticles();
        animateParticles();
        renderTimeline();
